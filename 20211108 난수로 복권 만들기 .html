Math를 활용한 구구단

        function randomInt(){
            return Math.floor(Math.random()*9)+1;
        }
        //구구단 문제 생성
        var ques = randomInt()+"*"+randomInt();
        //사용자로부터 답 입력
        var user = prompt(ques+"값은?");
    
        if(user == null){
            document.write("구구단 연습을 종료합니다")
        }else{
            var ans = eval(ques);
            if(ans == user){
                document.write("정답!
");
            }else{
                document.write("틀렸어요
");
            }
            document.write("정답은 "+ques+"="+""+ans+"입니다.
");
        }
