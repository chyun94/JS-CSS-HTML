//  함수의 호출과 선언, 더하는 방법

function adder(a,b){                        // 함수 작성
    var sum;
    sum = a+b;
    return sum;
}

var n2 = adder(123,456);                     // 함수 호출
console.log("123+456의 합은 " +n2+" 입니다."+"
